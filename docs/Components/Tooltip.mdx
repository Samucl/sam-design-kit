<hr/>

<p>Tooltip provides on-demand, contextual information to users as they hover over or focus on elements, enhancing understanding and interaction without cluttering the interface.</p>

<hr/>

<PropsDropdown>
  <tr>
    <th>Prop</th>
    <th>Type</th>
    <th>Default</th>
    <th>Description</th>
  </tr>
  <tr>
    <td>value</td>
    <td>string</td>
    <td><strong>required</strong></td>
    <td>Sets text for the Tooltip</td>
  </tr>
  <tr>
    <td>children</td>
    <td>React.ReactNode</td>
    <td><strong>required</strong></td>
    <td>Element which tooltip will appear from</td>
  </tr>
   <tr>
    <td>position</td>
    <td>top | right | bottom | left</td>
    <td>top</td>
    <td>Sets the position for Tooltip</td>
  </tr>
  <tr>
    <td>isDark</td>
    <td>boolean</td>
    <td>false</td>
    <td>The option for dark mode</td>
  </tr>
  <tr>
    <td>hidePointer</td>
    <td>boolean</td>
    <td>false</td>
    <td>Hides the Tooltip pointer</td>
  </tr>
</PropsDropdown>
