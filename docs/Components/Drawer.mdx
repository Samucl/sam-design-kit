<p>The Drawer component is a sliding panel that seamlessly integrates additional content or controls into the user experience, accessible with a simple gesture and easily dismissible to maintain focus on the main content.</p>

<hr/>

export const DrawerExample = (props) => {
  const { position, isDark } = props;
  const [isOpen, setIsOpen] = useState(false);
  const handleOpen = useCallback(() => setIsOpen(true), []);
  const handleClose = useCallback(() => setIsOpen(false), []);

  return (
    <>
      <div style={{ display: "flex", justifyContent: "center" }}>
        <StyledButton onClick={handleOpen}>
          {isOpen ? "Click to close Drawer" : "Click to open Drawer"}
        </StyledButton>
      </div>
      <Drawer isOpen={isOpen} onClose={handleClose} position={position} isDark={isDark}>
        <p>This is the Drawer content!</p>
      </Drawer>
    </>
  );
};
