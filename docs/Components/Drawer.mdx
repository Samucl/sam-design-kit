## Default (Right)

export const DrawerExample = (props) => {
  const { position } = props;
  const [isDrawerOpen, setDrawerOpen] = useState(false);
  return (
    <>
      <div style={{display: "flex", justifyContent: "center"}}><StyledLabel onClick={() => setDrawerOpen(true)}>Click to open Drawer</StyledLabel></div>
      <Drawer isOpen={isDrawerOpen} onClose={() => setDrawerOpen(false)} position={position}>
        <p>This is the Drawer content!</p>
      </Drawer>
    </>
  );
};
